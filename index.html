  4. Write a function which extracts a file extension. (desire 1: don't use regexp; desire 2: use only split method) 
    function extractFileExtension(path) {
// your code
}
extractFileExtension(); // ''
extractFileExtension(''); // ''
extractFileExtension(null); // ''
extractFileExtension('filename.ext'); // ext
extractFileExtension('folder/filename.png'); // png
extractFileExtension('folder/filename.png?logIn=true'); // png

    function extractFileExtension(path) {
	if (!path) { return ''; }

	let pathArray = path.split('/');
	let pathEnd = pathArray[pathArray.length - 1];

	return pathEnd.includes('?') ? pathEnd.split('?')[0].split('.')[1] : pathEnd.split('.')[1];
}
